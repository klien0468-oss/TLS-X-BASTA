local FloatEnabled = false
local NoclipEnabled = false
local InfiniteJumpEnabled = false
local ESPEnabled = false 
local SpeedEnabled = false
local WalkSpeedValue = 16

local NoclipConnection = nil
local FloatConnection = nil
local ESPConnection = nil 
local CharacterAddedConnection = nil
local Player = game.Players.LocalPlayer
local FloatToggle

_G.FriendColor = Color3.fromRGB(0, 0, 255)
_G.EnemyColor = Color3.fromRGB(255, 0, 0)
_G.UseTeamColor = true

local function getCharacter()
    local char = Player.Character
    if char and char.Parent then return char end
    return nil
end

local function setupFloat(char)
    local humanoid = char and char:FindFirstChildOfClass("Humanoid")
    if FloatEnabled and humanoid then
        humanoid:SetStateEnabled(Enum.HumanoidStateType.Jumping, false)
        humanoid:SetStateEnabled(Enum.HumanoidStateType.FallingDown, false)
        humanoid.PlatformStand = true
    end
end

local function enableFloat()
    if FloatEnabled then return end
    FloatEnabled = true
    local char = getCharacter()
    local humanoid = char and char:FindFirstChildOfClass("Humanoid")
    if not humanoid then
        Luna:Notification({ Title = "Lỗi", Icon = "error", ImageSource = "Material", Content = "Không tìm thấy Humanoid!" })
        FloatEnabled = false
        if FloatToggle then FloatToggle:SetValue(false) end
        return
    end
    humanoid:SetStateEnabled(Enum.HumanoidStateType.Jumping, false)
    humanoid:SetStateEnabled(Enum.HumanoidStateType.FallingDown, false)
    humanoid.PlatformStand = true
    FloatConnection = game:GetService("RunService").Heartbeat:Connect(function()
        local currentChar = getCharacter()
        if FloatEnabled and currentChar and currentChar.Parent then
            local hrp = currentChar:FindFirstChild("HumanoidRootPart")
            if hrp then hrp.Velocity = Vector3.new(hrp.Velocity.X, 0, hrp.Velocity.Z) end
        end
    end)
    if CharacterAddedConnection then CharacterAddedConnection:Disconnect() end
    CharacterAddedConnection = Player.CharacterAdded:Connect(function(newChar)
        task.wait(0.5) setupFloat(newChar)
    end)
    Luna:Notification({ Title = "Float", Icon = "flight", ImageSource = "Material", Content = "Đã bật Float" })
end

local function disableFloat()
    if not FloatEnabled then return end
    FloatEnabled = false
    if FloatConnection then FloatConnection:Disconnect() FloatConnection = nil end
    if CharacterAddedConnection then CharacterAddedConnection:Disconnect() CharacterAddedConnection = nil end
    local currentChar = getCharacter()
    local currentHumanoid = currentChar and currentChar:FindFirstChildOfClass("Humanoid")
    if currentHumanoid then
        currentHumanoid:SetStateEnabled(Enum.HumanoidStateType.Jumping, true)
        currentHumanoid:SetStateEnabled(Enum.HumanoidStateType.FallingDown, true)
        currentHumanoid.PlatformStand = false
    end
    Luna:Notification({ Title = "Float", Icon = "flight", ImageSource = "Material", Content = "Đã tắt Float" })
end

local function enableNoclip()
    if NoclipEnabled then return end
    NoclipEnabled = true
    local char = getCharacter()
    if char then
        for _, part in pairs(char:GetDescendants()) do
            if part:IsA("BasePart") then part.CanCollide = false end
        end
    end
    NoclipConnection = game:GetService("RunService").Stepped:Connect(function()
        if NoclipEnabled then
            local currentChar = getCharacter()
            if currentChar then
                for _, part in pairs(currentChar:GetDescendants()) do
                    if part:IsA("BasePart") and part.CanCollide == true then part.CanCollide = false end
                end
            end
        end
    end)
    Luna:Notification({ Title = "Noclip", Icon = "layers", ImageSource = "Material", Content = "Đã bật Noclip" })
end

local function disableNoclip()
    if not NoclipEnabled then return end
    NoclipEnabled = false
    if NoclipConnection then NoclipConnection:Disconnect() NoclipConnection = nil end
    local char = getCharacter()
    if char then
        for _, part in pairs(char:GetDescendants()) do
            if part:IsA("BasePart") then part.CanCollide = true end
        end
    end
    Luna:Notification({ Title = "Noclip", Icon = "layers", ImageSource = "Material", Content = "Đã tắt Noclip" })
end

local function bringAllPlayer()
    local targetPos = Player.Character and Player.Character:FindFirstChild("HumanoidRootPart")
    if not targetPos then
        Luna:Notification({ Title = "Lỗi", Icon = "error", ImageSource = "Material", Content = "Không tìm thấy vị trí nhân vật!" })
        return
    end
    local position = targetPos.Position
    local count = 0
    for _, otherPlayer in pairs(game.Players:GetPlayers()) do
        if otherPlayer ~= Player then
            local otherChar = otherPlayer.Character
            if otherChar and otherChar:FindFirstChild("HumanoidRootPart") then
                otherChar.HumanoidRootPart.CFrame = CFrame.new(position)
                count = count + 1
            end
        end
    end
    Luna:Notification({ Title = "Hoàn thành", Icon = "check", ImageSource = "Material", Content = "Đã kéo " .. count .. " người chơi!" })
end

game:GetService("UserInputService").JumpRequest:Connect(function()
    if InfiniteJumpEnabled then
        local char = getCharacter()
        local humanoid = char and char:FindFirstChildOfClass("Humanoid")
        if humanoid then humanoid:ChangeState("Jumping") end
    end
end)

task.spawn(function()
    while true do
        if SpeedEnabled then
            local char = Player.Character
            if char and char:FindFirstChildOfClass("Humanoid") then
                char:FindFirstChildOfClass("Humanoid").WalkSpeed = WalkSpeedValue
            end
        end
        task.wait(0.1)
    end
end)

local function applyESP(target, color)
    if target.Character and target.Character.Parent then
        local highlight = target.Character:FindFirstChild("GetReal")
        if not highlight then
            highlight = Instance.new("Highlight")
            highlight.RobloxLocked = true
            highlight.Name = "GetReal"
            highlight.Adornee = target.Character
            highlight.DepthMode = Enum.HighlightDepthMode.AlwaysOnTop
            highlight.FillColor = color
            highlight.Parent = target.Character
        else
            highlight.FillColor = color
        end
    end
end

local function removeESP()
    for _, v in pairs(game.Players:GetPlayers()) do
        if v.Character and v.Character:FindFirstChild("GetReal") then v.Character.GetReal:Destroy() end
    end
end

local ESPToggle = LocalPlayerTab:CreateToggle({
    Name = "Player ESP",
    Description = "Bật/tắt nhìn xuyên tường người chơi",
    CurrentValue = false,
    Callback = function(Value)
        ESPEnabled = Value
        if Value then
            ESPConnection = game:GetService("RunService").Heartbeat:Connect(function()
                if ESPEnabled then
                    for _, v in pairs(game.Players:GetPlayers()) do
                        if v ~= Player then
                            local finalColor = _G.UseTeamColor and v.TeamColor.Color or ((Player.TeamColor == v.TeamColor) and _G.FriendColor or _G.EnemyColor)
                            applyESP(v, finalColor)
                        end
                    end
                end
            end)
            Luna:Notification({ Title = "ESP", Icon = "visibility", ImageSource = "Material", Content = "Đã bật định vị!" })
        else
            if ESPConnection then ESPConnection:Disconnect() ESPConnection = nil end
            removeESP()
            Luna:Notification({ Title = "ESP", Icon = "visibility_off", ImageSource = "Material", Content = "Đã tắt ESP!" })
        end
    end
})
